library(tidytext)

df_ingredients <- df %>%
  unnest_tokens(word, Ingredients, token = "words") %>%
  count(word, sort = TRUE)

ggplot(df_ingredients %>% top_n(15), aes(x = reorder(word, n), y = n)) +
  geom_col(fill = "tomato") +
  coord_flip() +
  labs(title = "Most Common Ingredients in Global Street Food", x = "Ingredient", y = "Count") +
